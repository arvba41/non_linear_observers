clear; clc;

original system

u = @(t) 0;
fx = @(t,x) [ x(2) ;...
            -x(2) - sin(x(1)) + u(t)];

tspan = [0 4];
y0 = [pi/3; 0];
[t,y] = ode45(@(t,y) fx(t,y),tspan,y0);

observer design

y_val = @(tt) interp1(t,y(:,1),tt,'spline');
beta = 5;
K = [beta; 1];
fo = @(t,x) fx(t,x) + K * (y_val(t) - x(1));

y0 = [-pi/6; -pi/10];
[to,yo] = ode45(@(t,y) fo(t,y),tspan,y0);

error signals

yoo = interp1(t,y,to,'spline');
err = (yo - yoo);
